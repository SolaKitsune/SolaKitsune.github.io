// 自動顯示當前日期
const dateEl = document.getElementById('current-date');
if (dateEl) {
  const now = new Date();
  const year = now.getFullYear();
  const month = String(now.getMonth() + 1).padStart(2, '0');
  const day = String(now.getDate()).padStart(2, '0');
  dateEl.textContent = `${year}.${month}.${day}`;
}
